define cron::job (
  String         $command,
  String         $user     = 'root',
  Cron::Minute   $minute   = '*',
  Cron::Hour     $hour     = '*',
  Cron::Monthday $monthday = '*',
  Cron::Month    $month    = '*',
  Cron::Weekday  $weekday  = '*',
) {

  require ::cron

  cron { $title:
    ensure   => present,
    user     => $user,
    command  => $command,
    minute   => cron::prep4cron($minute),
    hour     => cron::prep4cron($hour),
    monthday => cron::prep4cron($monthday),
    month    => cron::prep4cron($month),
    weekday  => cron::prep4cron($weekday),
  }

}
